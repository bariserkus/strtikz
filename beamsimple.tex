\pgfkeys{
 /beamsimple/.is family, /beamsimple,
  default/.style={
  startx = 0cm,
  starty = 0cm,
  length = 4cm,
  line thickness = 2pt},
  startx/.store in =\startx,
  starty/.store in =\starty,
  endx/.store in =\endx,
  endy/.store in =\endy,
  line thickness/.store in =\linewidth,
  length/.store in =\length,
}


\newcommand{\beamsimple}[1][]{
\pgfkeys{/beamsimple, default, #1}

\tikzmath{
int \i;
real \startx, \starty;
real \length, \linewidth;
real \pinsupx, \rolsupx,
%Conversion all units to the unit pt
\startx = \startx;
\starty = \starty;
\linewidth = \linewidth;
\length = \length;
\pinsupx = 0.75*\linewidth;
\rolsupx = \length-0.75*\linewidth;
}
\begin{scope}[x=1pt, y=1pt, xshift = \startx, yshift=\starty, rotate=0];    % Draw everything in pt units
\draw [line width = \linewidth] (0,0) -- (\length,0);
%ToDo: Need to adjust support dimensions according to the beam dimensions and line thickness
\pinsupport[%
	triangle width=0.5cm,
	line width=1cm,
	line depth=0.2cm,
	line thickness=2pt,
	radius ratio = 0,
	x value = \pinsupx]
\rollersupport[%
	triangle width=0.5cm,
	line width=1cm,
	line depth=0.2cm,
	line thickness=2pt,
	radius ratio = 0,
	x value = \rolsupx]
\end{scope}
}
